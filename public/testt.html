// Movie constructor function function Movie(title, director, releaseYear) { this.title = title; this.director = director; this.releaseYear = releaseYear; this.reviews = []; // Array to store reviews // Method to add a review this.addReview = function(rating, comment) { const review = { rating: rating, comment: comment, date: new Date().toLocaleDateString() }; this.reviews.push(review); }; // Method to calculate average rating this.getAverageRating = function() { if (this.reviews.length === 0) return 0; const total = this.reviews.reduce((sum, review) => sum + review.rating, 0); return (total / this.reviews.length).toFixed(1); }; // Method to display movie info this.displayInfo = function() { console.log(`Title: ${this.title}`); console.log(`Director: ${this.director}`); console.log(`Release Year: ${this.releaseYear}`); console.log(`Average Rating: ${this.getAverageRating()}`); console.log(`Reviews:`); this.reviews.forEach(review => { console.log(` - [${review.date}] ${review.rating}/5: ${review.comment}`); }); }; } // Example usage const movie = new Movie("Inception", "Christopher Nolan", 2010); // Adding reviews movie.addReview(5, "Mind-blowing experience!"); movie.addReview(4, "Great visuals but a bit confusing."); movie.addReview(5, "A masterpiece!"); movie.displayInfo();
